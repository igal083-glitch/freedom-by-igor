export default async function handler(req, res) { const symbols = (req.query.symbols || "SPY,QQQM,VTI,VXUS,SPMO") .split(",") .map(s => s.trim().toUpperCase()); const key = process.env.FINNHUB_API_KEY; const results = await Promise.all( symbols.map(async symbol => { const url = `https://finnhub.io/api/v1/quote?symbol=${symbol}&token=${key}`; const r = await fetch(url); const q = await r.json(); return { symbol, price: q.c, previousClose: q.pc, change: q.d, changePercent: q.dp }; }) ); res.status(200).json({ data: results }); }
